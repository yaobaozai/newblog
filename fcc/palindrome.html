# 项目实战：回文检查器

  • 题目链接 (opens new window)

# 思路讲解

  1. 除去所有非字母数字的字符（标点、空格和符号）
  2. 使用双指针的方法，定义两个指针（其实就是索引），一个从前往后走，一个从后往前走，如果不相等则返回false
  3. 注意字母为大写的情况，统一转化成小写处理

# 代码实现

function palindrome(str) {
  // 首先去除特殊字符并切分成数组
  // 正则表达式中 []是表示里面的多个规则进行匹配 比如这里我们使用\W匹配字符，但是除了正常字母和数字以外的字符，还会包括下划线，我们需要再补上下划线也要去除，则使用[]
  str = str.replace(/[\W_]/g, '').split('');
  // 使用双指针的方法 一个从前往后走，一个从后往前走，如果不相等则返回false
  let left = 0, right = str.length - 1;
  while (left < right) {
    // 注意字母为大写的情况，统一转化成小写处理
    if (str[left].charCodeAt() >= 65 && str[left].charCodeAt() <= 90) {
      str[left] = str[left].toLowerCase();
    }
    // 同上
    if (str[right].charCodeAt() >= 65 && str[right].charCodeAt() <= 90) {
      str[right] = str[right].toLowerCase();
    }
    // 如果不相等则返回false
    if (str[left] !== str[right]) {
      return false;
    }
    // 左指针进位
    left++
    // 右指针进位
    right--
  }
  return true;
}
