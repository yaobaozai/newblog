# 项目实战：凯撒密码

  • 题目链接 (opens new window)

# 思路讲解

  • 这个题目相较项目实战中的其他题目来说就太基础了，考察charCodeAt String.fromCharCode API
  • 这里我使用map方法来完成，因为是要把字符串中每一项后移13位，首先需要把字符串切割成数组。
  • 这里只需要注意在后移13位以后，有可能超过了Z的90，那么需要重新从64开始算起，65是A。

# 代码实现

function rot13(str) { // LBH QVQ VG!
  // unicode编码65-90 表示A-Z
  return str.split('').map(item => {
    const itemCode = item.charCodeAt(item);
    if (itemCode >= 65 && itemCode <= 90) {
      const movedCode = itemCode + 13;
      if (movedCode > 90) {
        return String.fromCharCode(movedCode - 90 + 64)
      }
      return String.fromCharCode(movedCode)  
    } else {
      return item
    }
  }).join("");
}
