# 项目实战：收银机

  • 题目链接 (opens new window)

# 思路讲解

  • 这个题目的意思就是我们生活中的找零钱，优先把大额的找给人家，比如 需要找零102，然后应该是1张100加两张1块，如果没有100的，然后就找50的，50的也没有，就找20的。当然这里是没有50这个单位的，只有20的。搞清楚要做什么以后，基本就了解了，只不过是用代码表示找钱的这个过程。
  • 首先我们需要创建一个单位对象moneyMap，方便我们在遍历收银机中的钱，也就是cid参数，它是一个长这样的二维数组[["PENNY", 0.01], ["NICKEL", 0], ["DIME", 0], ["QUARTER", 0], ["ONE", 1], ["FIVE", 0], ["TEN", 0], ["TWENTY", 0], ["ONE HUNDRED", 0]]，有了这两个以后，我们就可以知道对应的每个面值，我们有多少张了，比如["ONE HUNDRED", 500]，我们知道1张是100，所以是5张。
  • 从后往前遍历cid，因为是从大的面值开始找钱动作。在每一次找钱的动作里，我们需要确保，收银机里有对应面额的钱，并且需要找的钱是大于当前遍历到的面额的。
  • 由于我们一个面额可能需要多次找零，比如找500，5张100，所以是一个循环套循环的结构，外层循环表示当前的面额，内层循环则是具体找零，完成当前面额的找零再进入下一个面额的找零。

# 注意

  • javascript浮点数计算问题，所以我们这里统一乘以100，最后把结果除以100
  • JSON.parse(JSON.stringify(Object)) 是常用的深拷贝方法，这里用slice拷贝数组的话是浅拷贝，使用slice进行浅拷贝的话，由于cid是一个二维数组，修改里面的子数组，那么我copy出来的数据还是会被改变，所以得用深拷贝确保数组中的数组/对象，对象中的数组/对象，依然是复制出来的。

# 代码实现

function checkCashRegister(price, cash, cid) {
    // 创建一个单位对象moneyMap，方便我们后续遍历收银机中的钱时能知道对应面额有多少张
  const moneyMap = {
    "PENNY": 1,
    "NICKEL": 5,
    "DIME": 10,
    "QUARTER": 25,
    "ONE": 100,
    "FIVE": 500,
    "TEN": 1000,
    "TWENTY": 2000,
    "ONE HUNDRED": 10000
  }
  // 需要找的金额，由于在每次找零时我们会减这个数字，所以需要两个变量，一个用来计算，一个用来比较  
  const diffChange = (cash - price) * 100
  let needChange = (cash - price) * 100;
  // 当刚好可以找零时，也就是对应题目中的CLOSED状态，我们需要把cid原样返回，但我会在找零动作中修改cid，所以需要copy一份，由于是二维数组，所以需要用深拷贝。   
  const copyCid = JSON.parse(JSON.stringify(cid));
  // 找零结果
  let resultArr = [];
  // 收银机中的总额
  let total = 0;
  // 从大到小遍历cid即收银机中的钱
  for (let len = cid.length, i = len - 1; i >= 0; i--) {
    // 计算收银机中的总额
    total += cid[i][1] * 100;
    // 确保可以找钱 需要找的钱比当前面额大，并且当前面额不为0
    if (needChange >= moneyMap[cid[i][0]] && cid[i][1]) {
      // 统计这个面额找了多少钱
      let waitMoney = 0;
      // 再进行具体找钱的动作
      while (needChange >= moneyMap[cid[i][0]] && cid[i][1]) {
        // 每找一次，收银机中对应面额的钱则需要减少一次
        cid[i][1] = (cid[i][1] * 100 - moneyMap[cid[i][0]]) / 100;
        // 每找一次，统计一次这个面额找的钱
        waitMoney += moneyMap[cid[i][0]];
        // 每找一次，剩余多少钱待找
        needChange -= moneyMap[cid[i][0]];
      }
      // 完成当前面额找零，保存结果
      resultArr.push([cid[i][0], waitMoney / 100])
    }
  }
  // 刚好找零
  if (diffChange - total === 0) {
    return { status: "CLOSED", change: copyCid }
    // 经过找零以后，待找的钱还是大于0，说明我们找不开。
  } else if (needChange > 0) {
    return { status: "INSUFFICIENT_FUNDS", change: [] }
  } else {
    // 能完成找零，还有剩余的钱
    return { status: "OPEN", change: resultArr }
  }
}
