function whatIsInAName(collection, source) {
    const srcKeys = Object.keys(source);
    return collection.filter(function(obj) {
        for (let i = 0; i < srcKeys.length; i++) {
            if (!obj.hasOwnProperty(srcKeys[i]) || obj[srcKeys[i]] !== source[srcKeys[i]]) {
                return false;
            }
        }
        return true;
    });
}

# 算法中级：儿童黑话

  • 题目链接 (opens new window)

  • 利用正则表达式

function translatePigLatin(str) {
    let pigLatin = "";
    const regex = /[aeiou]/gi;
    if (str[0].match(regex)) {
        pigLatin = str + "way";
    } else if (str.match(regex) === null) {
        pigLatin = str + "ay";
    } else {
        // 注意match方法返回值是一个存放匹配结果的数组。
        const vowelIndice = str.indexOf(str.match(regex)[0]);
        // 初学者可以在这复习一下substr 与 substring的区别，substr第二个参数是代表length 而 substring 第二个参数是endIndex
        pigLatin = str.substr(vowelIndice) + str.substr(0, vowelIndice) + "ay";
    }
    return pigLatin;
}

function myReplace(str, before, after) {
    // 考察字符串方法运用
    const index = str.indexOf(before);
    if (str[index] === str[index].toUpperCase()) {
        after = after.charAt(0).toUpperCase() + after.slice(1);
    } else {
        after = after.charAt(0).toLowerCase() + after.slice(1);
    }
    str = str.replace(before, after);
    return str;
}

function pairElement(str) {
    // 用一个新数组来存
    let paired = [];
    const search = function(char) {
        switch (char) {
            case "A":
                paired.push(["A", "T"]);
                break;
            case "T":
                paired.push(["T", "A"]);
                break;
            case "C":
                paired.push(["C", "G"]);
                break;
            case "G":
                paired.push(["G", "C"]);
                break;
        }
    };
    for (let i = 0; i < str.length; i++) {
        search(str[i]);
    }
    return paired;
}

function fearNotLetter(str) {
    for (let i = 0; i < str.length; i++) {
        // 获取当前循环到字母对应的编码
        const code = str.charCodeAt(i);
        // 如果当前编码对不上
        if (code !== str.charCodeAt(0) + i) {
            return String.fromCharCode(code - 1);
        }
    }
    return undefined;
}

# 算法中级：集合排序

  • 题目链接 (opens new window)
function uniteUnique(arr) {
    // arguments类数组转换成真正的数组，便于取值
    const args = [...arguments];
    let result = [];
    // 朴实无华的双层循环
    for (let i = 0; i < args.length; i++) {
        for (let j = 0; j < args[i].length; j++) {
            if (!result.includes(args[i][j])) {
                result.push(args[i][j]);
            }
        }
    }
    return result;
}

# 算法中级：转换 HTML 实体

  • 题目链接 (opens new window)

# 解法一

function convertHTML(str) {
    var temp = str.split("");
    // 工作中常用switch来转换多种情况
    for (var i = 0; i < temp.length; i++) {
        switch (temp[i]) {
            case "<":
                temp[i] = "&lt;";
                break;
            case "&":
                temp[i] = "&amp;";
                break;
            case ">":
                temp[i] = "&gt;";
                break;
            case '"':
                temp[i] = "&quot;";
                break;
            case "'":
                temp[i] = "&apos;";
                break;
        }
    }
    temp = temp.join("");
    return temp;
}

function convertHTML(str) {
    // 利用对象、replace特性 注意replace的第二个参数 - 函数的回调值，是匹配到的对应情况
    const htmlEntity = {
        "&": "&amp;",
        "<": "&lt;",
        ">": "&gt;",
        '"': "&quot;",
        "'": "&apos;",
    };
    return str.replace(/[&<>'"]/g, (match) => htmlEntity[match]);
}

# 算法中级：求斐波那契数组中的奇数之和

  • 题目链接 (opens new window)
function sumFibs(num) {
    // 利用数组方法filter与reduce
    let fibArr = [1, 1];
    let nextNum = 0;
    while ((nextNum = fibArr[0] + fibArr[1]) <= num) {
        fibArr.unshift(nextNum);
    }
    return fibArr.filter((item) => item % 2 !== 0).reduce((prev, current) => prev + current);
}

# 算法中级：对所有素数求和

  • 题目链接 (opens new window)
function sumPrimes(num) {
    // 封装一个方法，检查是否是素数
    function isPrime(num) {
        for (let i = 2; i <= Math.sqrt(num); i++) {
            if (num % i == 0) return false;
        }
        return true;
    }

    // 对所有素数求和
    let sum = 0;
    for (let i = 2; i <= num; i++) {
        if (isPrime(i)) sum += i;
    }
    return sum;
}

# 算法中级：最小公倍数

  • 题目链接 (opens new window)
function smallestCommons(arr) {
    // 按升序排序
    const [min, max] = arr.sort((a, b) => a - b);
    // 构造包含范围内所有数字的数组
    const range = Array(max - min + 1)
        .fill(0)
        .map((_, i) => i + min);
    // 使用辗转相除法求两个数的最大公约数
    const gcd = (a, b) => (b === 0 ? a : gcd(b, a % b));
    // 使用公式法求两个数的最小公倍数
    const lcm = (a, b) => (a * b) / gcd(a, b);
    // 多个数的最小公倍数
    return range.reduce((multiple, curr) => lcm(multiple, curr));
}

    const index = arr.findIndex((item) => {
        return func(item);
    });
    return index === -1 ? [] : arr.slice(index);

# 算法中级：扁平化

  • 题目链接 (opens new window)
function steamrollArray(arr) {
    let flatArr = [];
    for (let i = 0; i < arr.length; i++) {
        if (Array.isArray(arr[i])) {
            flatArr.push(...steamrollArray(arr[i]));
        } else {
            flatArr.push(arr[i]);
        }
    }
    return flatArr;
}

# 算法中级：可选参数

  • 题目链接 (opens new window)
function addTogether() {
    const [first, second] = arguments;
    if (typeof first !== "number") return undefined;
    if (!second) {
        return function(number) {
            if (typeof number !== "number") return undefined;
            return first + number;
        };
    }
    if (second && typeof second !== "number") return undefined;
    return first + second;
}

# 算法中级：构造一个 Person 类

  • 题目链接 (opens new window)
const Person = function(firstAndLast) {
    let _firstName, _lastName;
    this.getFullName = function() {
        return _firstName + " " + _lastName;
    };
    this.getFirstName = function() {
        return _firstName;
    };
    this.getLastName = function() {
        return _lastName;
    };
    this.setFirstName = function(name) {
        _firstName = name;
    };
    this.setLastName = function(name) {
        _lastName = name;
    };
    this.setFullName = function(name) {
        name = name.split(" ");
        _firstName = name[0];
        _lastName = name[1];
    };
    this.setFullName(firstAndLast);
};

function orbitalPeriod(arr) {
    const GM = 398600.4418;
    const earthRadius = 6367.4447;
    return arr.map((item) => {
        return {
            name: item.name,
            orbitalPeriod: Math.round(2 * Math.PI * Math.sqrt(Math.pow(earthRadius + item.avgAlt, 3) / GM)),
        };
    });
}
