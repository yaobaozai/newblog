# 项目实战：罗马数字转换器

  • 题目链接 (opens new window)

# 思路讲解

  • 这个题目思路很简单，看测试用例，最高我们需要对3999进行转换，那么我可以完全把百位0-9，十位0-9，个位0-9对应的罗马数字找出来，再来个拼接就可以了，注意规律，千位直接拼接M。

# 代码实现

function convertToRoman(num) {
    const map = {
        0: '',
        1: 'I',
        2: 'II',
        3: 'III',
        4: 'IV',
        5: 'V',
        6: 'VI',
        7: 'VII',
        8: 'VIII',
        9: 'IX',
        10: 'X',
        20: 'XX',
        30: 'XXX',
        40: 'XL',
        50: 'L',
        60: 'LX',
        70: 'LXX',
        80: 'LXXX',
        90: 'XC',
        100: 'C',
        200: 'CC',
        300: 'CCC',
        400: 'CD',
        500: 'D',
        600: 'DC',
        700: 'DCC',
        800: 'DCCC',
        900: 'CM'
    };
    let a = Math.floor(num / 1000);
    const b = Math.floor((num % 1000) / 100) * 100;
    const c = Math.floor((num % 100) / 10) * 10;
    const d = num % 10;
    let tho = '';
    const han = map[b];
    const ten = map[c];
    const bit = map[d];
    while (a > 0) {
        tho += 'M';
        a--;
    }
    return tho + han + ten + bit;
}
