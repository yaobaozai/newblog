var lengthOfLongestSubstring = function(s) {
    if (!s.length) return 0;
    if (!s.trim()) return 1;
    let lenArr = [];
    for (let i = 0; i < s.length; i++) {
        let str = "";
        for (let n = i; n < s.length; n++) {
            if (str.indexOf(s[n]) === -1) {
                str += s[n];
                if (n + 1 === s.length) {
                    lenArr.push(str.length);
                }
            } else {
                lenArr.push(str.length);
                break;
            }
        }
    }
    return Math.max(...lenArr);
};
